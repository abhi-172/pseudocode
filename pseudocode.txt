READ A,B

//Procedure to Add Two Numbers
PROCEDURE ADD(X:INTEGER,Y:INTEGER)
   DECLARE SUM:INTEGER
   SET SUM = X + Y
   PRINT "Sum =", SUM
END PROCEDURE

//Function to Add Two Numbers
FUNCTION ADD(X:INTEGER,Y:INTEGER):INTEGER
   RETURN   X + Y
END FUNCTION
PRINT ADD(5,89)

//Function to Find Area of Triangle
FUNCTION TRIANGLE_AREA(BASE:REAL,HEIGHT:REAL):REAL
   RETURN 0.5 * BASE * HEIGHT
END FUNCTION

PRINT TRIANGLE_AREA(10,5)

//Function to Find Area of Circle
FUNCTION CIRCLE_AREA(R:INTEGER):REAL
   RETURN 3.14 * R * R
END FUNCTION

PRINT CIRCLE_AREA(7)

//Function to Find Greatest of Two Numbers
FUNCTION MAX(A:INTEGER,B:INTEGER):INTEGER
   IF A>B THEN
      RETURN A
   ELSE
      RETURN B
   END IF
END FUNCTION

PRINT MAX(10,25)

//Function to Find Absolute Value
FUNCTION ABS_VAL(N:INTEGER):INTEGER
   IF N<0 THEN
      RETURN -N
   ELSE
      RETURN N
   END IF
END FUNCTION

PRINT ABS_VAL(-8)

//Function to Find Factorial
FUNCTION FACT(N:INTEGER):INTEGER
   DECLARE I,F:INTEGER
   SET F = 1

   FOR I ← 1 TO N
      SET F = F * I
   END FOR

   RETURN F
END FUNCTION

PRINT FACT(5)

//Function to Find Sum of Series 1 to N
FUNCTION SUM_SERIES(N:INTEGER):INTEGER
   DECLARE I,SUM:INTEGER
   SET SUM = 0

   FOR I ← 1 TO N
      SET SUM = SUM + I
   END FOR

   RETURN SUM
END FUNCTION

PRINT SUM_SERIES(10)


//Function to Find  Series Sum
FUNCTION SeriesSum(N:INTEGER):REAL
   DECLARE I:INTEGER
   DECLARE SUM:REAL
   SET SUM = 0

   FOR I ← 1 TO N
      SET SUM = SUM + 1/I
   END FOR

   RETURN SUM
END FUNCTION

PRINT SeriesSum(5)

//Function to Reverse Number
FUNCTION REVERSE(N:INTEGER):INTEGER
   DECLARE REV,DIGIT:INTEGER
   SET REV = 0

   WHILE N>0
      SET DIGIT = N MOD 10
      SET REV = REV*10 + DIGIT
      SET N = N / 10
   END WHILE

   RETURN REV
END FUNCTION

PRINT REVERSE(1234)


//Function to Check Palindrome
FUNCTION PALINDROME(N:INTEGER):BOOLEAN
   IF N = REVERSE(N) THEN
      RETURN TRUE
   ELSE
      RETURN FALSE
   END IF
END FUNCTION

PRINT PALINDROME(121)
